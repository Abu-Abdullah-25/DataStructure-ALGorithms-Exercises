    bool isAnagram(string s, string t) {
        if(t.length() != t.length())
            return false;

        map<char,int> myMap;
        for(int i = 0; i < s.length(); i++)
        {
            myMap[s[i]] += 1;
        }

        for(int i = 0; i < t.length(); i++)
        {
            myMap[t[i]] -= 1;
        }

        for(const auto& pair : myMap)
        {
            if(pair.second != 0)
                return false;
        }

        return true;
    }
