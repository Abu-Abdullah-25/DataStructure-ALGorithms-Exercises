{

int MyFloor(float Number)
{
    int intPart;
    intPart = int(Number);

    float FractionsPart = GetFractionPart(Number);

    if (abs(FractionsPart) >= 0.1)
    {
        if (Number > 0)

            return intPart;
        else
            return --intPart;
    }

    else
        return intPart;

}
