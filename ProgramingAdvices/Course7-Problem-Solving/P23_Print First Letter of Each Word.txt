string ReadString() {

    cout << "Please enter your string: \n";
    getline(cin, S1);
    return S1;
}

void PrintFirstLetterOfEachWord(string S1) {
    bool isFirstLetter = true;
    cout << "\nFirst letters of this string:\n";
    for (int i = 0; i < S1.length(); i++) {
        if (isFirstLetter && S1[i] != ' ') {
            cout << S1[i] << endl;
        }
        isFirstLetter = (S1[i] == ' ');
    }
}
